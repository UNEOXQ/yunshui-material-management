const API_BASE_URL = 'http://192.168.68.95:3004/api';

// API æœå‹™å‡½æ•¸
const apiService = {
  // å–å¾—åŸºæåˆ—è¡¨
  getMaterials: async () => {
    try {
      const response = await fetch(`${API_BASE_URL}/materials`);
      const data = await response.json();
      console.log('Materials response:', data);
      
      // ç¢ºä¿å›å‚³é™£åˆ—
      if (data.success && Array.isArray(data.data)) {
        return data.data;
      } else if (Array.isArray(data)) {
        return data;
      } else {
        return [];
      }
    } catch (error) {
      console.error('å–å¾—åŸºæå¤±æ•—:', error);
      return [];
    }
  },

  // å–å¾—è¨‚å–®åˆ—è¡¨
  getOrders: async () => {
    try {
      const response = await fetch(`${API_BASE_URL}/orders`);
      const data = await response.json();
      console.log('Orders response:', data);
      
      // ç¢ºä¿å›å‚³é™£åˆ—
      if (data.success && Array.isArray(data.data)) {
        return data.data;
      } else if (Array.isArray(data)) {
        return data;
      } else {
        return [];
      }
    } catch (error) {
      console.error('å–å¾—è¨‚å–®å¤±æ•—:', error);
      return [];
    }
  },

  // æ–°å¢åŸºæ
  createMaterial: async (material) => {
    try {
      const response = await fetch(`${API_BASE_URL}/materials`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(material),
      });
      const data = await response.json();
      return data;
    } catch (error) {
      console.error('æ–°å¢åŸºæå¤±æ•—:', error);
      return { success: false, error: error.message };
    }
  },
};

      console.log('Loaded orders:', ordersData);
    } catch (error) {
      console.error('è¼‰å…¥è³‡æ–™éŒ¯èª¤:', error);
      Alert.alert('éŒ¯èª¤', 'è¼‰å…¥è³‡æ–™å¤±æ•—ï¼Œè«‹æª¢æŸ¥å¾Œç«¯æœå‹™æ˜¯å¦é‹è¡Œ');
